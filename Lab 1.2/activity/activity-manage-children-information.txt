@startuml
|User|
start

:User selects to manage children information;
|System|
:Retrieve children information for the user from the database;
:Display children information to the user;
|User|
if (User wants to search children) then (Search)
  |System|
  :User provides search criteria;
  :Retrieve filtered children information from the database;
  :Display search results to the user;
else (Create/Read/Update/Delete)
  if (User chooses to create child) then (Create)
    |User|
    :User provides child information;
    |System|
    :Validate child information;
    |Database|
    :Create a new child record in the database;
    :Notify the user of successful creation;
  endif
  if (User selects a child to view) then (Read)
    |User|
    :User selects a child to view;
    |System|
    :Retrieve the selected child's information from the database;
    :Display the child's information to the user;
  endif
  if (User selects a child to update) then (Update)
    |User|
    :User selects a child to update;
    |System|
    :Retrieve the selected child's information from the database;
    :Display the child's information to the user;
    :User updates child information;
    |System|
    :Validate updated child information;
    |Database|
    :Update the child's information in the database;
    :Notify the user of successful update;
  endif
  if (User selects a child to delete) then (Delete)
    |User|
    :User selects a child to delete;
    |System|
    :Retrieve the selected child's information from the database;
    |Database|
    :Delete the selected child's information from the database;
    :Notify the user of successful deletion;
  endif
endif

|User|
stop
@enduml
